def binarysearch(l,element):
  first=0
  last=len(l)-1
  found=False
  while first<=last and not found:
    midpoint = (first + last)//2
    if l[midpoint] == element:
	    found = True
    else:
	    if element < l[midpoint]:
	      last = midpoint-1
	    else:
	      first = midpoint+1
  string="Element found at {}".format(midpoint)
  if(first>last):
    return "Element not found"
  else: return string
